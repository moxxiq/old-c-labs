345 int *go_zigzag(int **arr, int N, int pos){
346  // pos [0;N^2]
347  int i,j,inner_i,inner_j,k = 0;
348  i = 0;
349  while (i < 2*N-1){
350  if (i-N < 0){//зміна індексу і відносно головної діагоналі
351  inner_i = 0;
352  } else{
353  inner_i = i-(N-1);
354  }
355  j = (N-1)-i+inner_i;
356  inner_j = 0;
357  while (j<N-inner_i){
358  if (k == pos){
359  return &arr[inner_i+inner_j][j];
360  }
361  inner_j++;
362  j++;
363  k++;
364  }
365  i++;
366  }
367 }
